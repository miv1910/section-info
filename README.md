# skat-get-section-info

## run application

### command `run` args
    Name:     "log-level",
    Usage:    "warn",
    EnvVars:  []string{"SECTIONS_LOG_LEVEL"},
    Required: true,
    Value:    "warn",
			
    Name:     "listen-addr",
    EnvVars:  []string{"SECTIONS_LISTEN_ADDR"},
    Required: true,
    Value:    ":8080",
    
    Name:     "db-driver-name",
    EnvVars:  []string{"SECTIONS_DB_DRIVER"},
    Required: true,
    Value:    "mssql",

    Name:     "db-connection-string",
    EnvVars:  []string{"SECTIONS_DB_CONNECTION"},
    Required: true,
    Value:    "server=192.168.8.250;port=1433;user id=As;password=queue;database=dbqueue_clerkbot",

## work with docker file 
### build docker image
docker build -t section-info .

### run docker image
docker run -p 8080:8080 -t --env-file ./env.list section-info
